}

export function getRunner(runnerId: string): Runner | undefined {
  for (const race of ALL_RACES) {
    const r = race.runners.find(r => r.id === runnerId);
    if (r) return r;
  }
  return undefined;
}

export function offTimeToMinutes(offTime: string): number {
  const [h, m] = offTime.split(":").map(Number);
  return h * 60 + m;
}

export function getNextRace(): Race | undefined {
  const todayRaces = getRacesForDay(CURRENT_DAY_INDEX);
  return todayRaces.find(r => r.status === "declared") || todayRaces.find(r => r.status === "off");
}

export function getRaceById(id: string): Race | undefined {
  return ALL_RACES.find(r => r.id === id);

function useCountdown(targetMinutes: number) {
  const [remaining, setRemaining] = useState(() => {
    const diff = targetMinutes - SIMULATED_NOW_MINUTES;
    return Math.max(0, diff);
  });

  useEffect(() => {
    if (remaining <= 0) return;
    const id = setInterval(() => {
      setRemaining(prev => Math.max(0, prev - 1 / 60));
    }, 1000);
    return () => clearInterval(id);
  }, []);

  const mins = Math.floor(remaining);
  const secs = Math.floor((remaining - mins) * 60);
  return `${String(mins).padStart(2, "0")}:${String(secs).padStart(2, "0")}`;
}

export default function Racecard() {

  const [activeRaceId, setActiveRaceId] = useState(defaultRaceId);

  const [localPicks, setLocalPicks] = useState<Record<string, string>>(() => {
    const map: Record<string, string> = {};
    PICKS.filter(p => p.userId === CURRENT_USER_ID).forEach(p => { map[p.raceId] = p.runnerId; });
    return map;
  });

    if (!activeRace || isLocked) return;

      if (prev[activeRace.id] === runnerId) { const next = { ...prev }; delete next[activeRace.id]; return next; }

                ? <span style={{ fontSize: "12px", fontWeight: active ? 700 : 500 }} className={active ? "text-green-900" : "text-gray-500"}>Overall</span>
                : <>
                    <span style={{ fontSize: "9px", fontWeight: 600, textTransform: "uppercase" }} className={active ? "text-green-700" : "text-gray-400"}>
                      {i === CURRENT_DAY_INDEX ? "Today" : `Day ${i + 1}`}
                    </span>
                    <span style={{ fontSize: "11px", fontWeight: active ? 700 : 500 }} className={active ? "text-green-900" : "text-gray-500"}>
                      {tab.split(" ")[0]}
                    </span>
                  </>}
            </button>
          );
        })}
      </div>

      {/* Content */}
      <div className="p-3 flex flex-col gap-2">
        {isOverall ? (
          <OverallSummary userId={selectedUserId} />
        ) : races.length === 0 ? (
          <p className="text-center text-gray-400 py-8" style={{ fontSize: "14px" }}>No races for this day.</p>
        ) : (
          <>
            <div className="grid items-center px-3 py-2 bg-green-900 rounded-t-lg" style={{ gridTemplateColumns: "4.5rem 1fr 4rem 4.5rem" }}>
              {["Time","Horse","Result","Points"].map((h, i) => (
                <span key={h} className={`text-green-300 ${i >= 2 ? "text-center" : ""}`}
                  style={{ fontSize: "10px", fontWeight: 700, textTransform: "uppercase" }}>{h}</span>
              ))}
            </div>
            <div className="bg-white rounded-b-lg overflow-hidden shadow-sm">
              {races.map((race, idx) => (
                <RaceRow key={race.id} race={race} userId={selectedUserId} isViewingSelf={isViewingSelf} isEven={idx % 2 === 0} />
              ))}
            </div>
          </>
        )}
      </div>
    </div>
  );
}

function RaceRow({ race, userId, isViewingSelf, isEven }: {
  race: Race; userId: string; isViewingSelf: boolean; isEven: boolean;
}) {
  const raceOffMinutes = offTimeToMinutes(race.offTime);
  const raceIsOff = SIMULATED_NOW_MINUTES >= raceOffMinutes;
  const isResult = race.status === "result";
  const hideFromOther = !raceIsOff && !isViewingSelf && !isResult;
  const minsToOff = raceOffMinutes - SIMULATED_NOW_MINUTES;
  const pick = getUserPick(userId, race.id);
  const runner = pick ? getRunner(pick.runnerId) : undefined;

  return (
    <div className={`grid items-center border-b border-gray-100 last:border-0 ${isEven ? "bg-white" : "bg-gray-50/40"}`}
      style={{ gridTemplateColumns: "4.5rem 1fr 4rem 4.5rem" }}>
      <div className="flex flex-col items-start px-3 py-3">
        <span className="text-green-900" style={{ fontSize: "13px", fontWeight: 700 }}>{race.offTime}</span>
        <span className={`w-1.5 h-1.5 rounded-full mt-1 ${isResult ? "bg-green-500" : raceIsOff ? "bg-red-500 animate-pulse" : "bg-gray-300"}`} />
      </div>

      <div className="flex items-center gap-2 py-3 px-1 min-w-0 border-l border-gray-100">
        {hideFromOther ? (
          <div className="flex items-center gap-2 text-gray-400">
            <EyeOff size={14} className="flex-shrink-0" />
            <div className="min-w-0">
              <span className="text-gray-400" style={{ fontSize: "12px", fontWeight: 600 }}>Hidden</span>
              {minsToOff > 0 && (
                <div className="flex items-center gap-1 mt-0.5">
                  <Clock size={10} className="text-gray-400" />
                  <span className="text-gray-400" style={{ fontSize: "10px" }}>Off in {minutesToHHMM(minsToOff)}</span>
                </div>
              )}
            </div>
          </div>
        ) : runner ? (
          <>
            <SilkIcon colors={runner.silkColors} size={24} />
            <div className="min-w-0">
              <p className="text-gray-900 truncate" style={{ fontSize: "12px", fontWeight: 600 }}>{runner.horseName}</p>
              <p className="text-gray-400 truncate" style={{ fontSize: "10px" }}>{runner.trainer}</p>
            </div>
          </>
        ) : <span className="text-gray-400" style={{ fontSize: "12px" }}>No pick</span>}
      </div>

      <div className="flex items-center justify-center py-3 border-l border-gray-100 px-1">
        {hideFromOther || !runner ? (
          <span className="text-gray-300" style={{ fontSize: "11px" }}>—</span>
        ) : isResult ? (
          <span className="px-1.5 py-0.5 rounded text-white"
            style={{ fontSize: "11px", fontWeight: 700,
              backgroundColor: runner.finishPosition === 1 ? "#c9a227" :
                runner.finishPosition && runner.finishPosition <= 3 ? "#16a34a" :
                runner.finishStatus && runner.finishStatus !== "finished" ? "#dc2626" : "#6b7280" }}>
            {finishLabel(runner)}
          </span>
        ) : raceIsOff ? (
          <span className="text-red-500" style={{ fontSize: "11px", fontWeight: 700 }}>Off</span>
        ) : <span className="text-gray-400" style={{ fontSize: "11px" }}>–</span>}
      </div>

      <div className="flex items-center justify-center py-3 border-l border-gray-100 px-1">
        {hideFromOther || !runner ? (
          <span className="text-gray-300" style={{ fontSize: "11px" }}>—</span>
        ) : isResult && runner.pointsAwarded !== undefined ? (
          <span style={{ fontSize: "13px", fontWeight: 700,
            color: runner.pointsAwarded > 0 ? "#16a34a" : runner.pointsAwarded < 0 ? "#dc2626" : "#9ca3af" }}>
            {runner.pointsAwarded > 0 ? `+${runner.pointsAwarded}` : runner.pointsAwarded === 0 ? "0" : `${runner.pointsAwarded}`}
          </span>
        ) : <span className="text-gray-300" style={{ fontSize: "11px" }}>—</span>}
      </div>
    </div>
  );
}

    </div>
  );
}

Codex Integration Notes
When wiring up to the live backend, these are the key substitutions to make in mockData.ts:

Mock constant	Replace with
ALL_RACES	GET /api/races?day=:dayIndex
PICKS	GET /api/picks?userId=:id
LEADERBOARD	GET /api/leaderboard (poll ≤15s)
USERS	GET /api/users or from auth session
CURRENT_USER_ID	Auth session user.id
CURRENT_DAY_INDEX	Derived from meeting.date vs Date.now() UTC
SIMULATED_NOW_MINUTES	new Date().getUTCHours() * 60 + new Date().getUTCMinutes()
Pick submission	POST /api/picks { user_id, race_id, runner_id }
The pick lock check (SIMULATED_NOW_MINUTES >= offTimeToMinutes) in Racecard.tsx is cosmetic only — the server enforces the real lock via off_time per the architecture lock document.
